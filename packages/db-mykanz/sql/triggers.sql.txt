-- ==========================================
-- 1. AUTO UPDATE updated_at
-- Trigger generik untuk semua tabel yang punya kolom updated_at
-- ==========================================

CREATE OR REPLACE FUNCTION trigger_set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Apply ke semua tabel yang relevan
CREATE TRIGGER set_updated_at_wallets
  BEFORE UPDATE ON wallets
  FOR EACH ROW EXECUTE FUNCTION trigger_set_updated_at();

CREATE TRIGGER set_updated_at_categories
  BEFORE UPDATE ON categories
  FOR EACH ROW EXECUTE FUNCTION trigger_set_updated_at();

CREATE TRIGGER set_updated_at_fiat_transactions
  BEFORE UPDATE ON fiat_transactions
  FOR EACH ROW EXECUTE FUNCTION trigger_set_updated_at();

CREATE TRIGGER set_updated_at_budgets
  BEFORE UPDATE ON budgets
  FOR EACH ROW EXECUTE FUNCTION trigger_set_updated_at();

CREATE TRIGGER set_updated_at_goals
  BEFORE UPDATE ON goals
  FOR EACH ROW EXECUTE FUNCTION trigger_set_updated_at();

CREATE TRIGGER set_updated_at_assets
  BEFORE UPDATE ON assets
  FOR EACH ROW EXECUTE FUNCTION trigger_set_updated_at();

CREATE TRIGGER set_updated_at_user_portfolios
  BEFORE UPDATE ON user_portfolios
  FOR EACH ROW EXECUTE FUNCTION trigger_set_updated_at();

CREATE TRIGGER set_updated_at_asset_transactions
  BEFORE UPDATE ON asset_transactions
  FOR EACH ROW EXECUTE FUNCTION trigger_set_updated_at();

-- ==========================================
-- 2. UPDATE PORTFOLIO STATS (DCA Kalkulasi)
-- Otomatis re-kalkulasi total_units dan average_buy_price
-- di tabel user_portfolios setiap ada INSERT / UPDATE / DELETE
-- di tabel asset_transactions
--
-- Formula DCA (Dollar Cost Averaging):
--   total_units       = SUM(units) dari semua BELI + SALDO_AWAL - SUM(units) dari JUAL
--   average_buy_price = SUM(total_amount BELI + SALDO_AWAL) / SUM(units BELI + SALDO_AWAL)
--
-- Logic:
--   - BELI & SALDO_AWAL menambah units dan mempengaruhi DCA
--   - JUAL mengurangi units tapi TIDAK mengubah average_buy_price
-- ==========================================

CREATE OR REPLACE FUNCTION update_portfolio_stats()
RETURNS TRIGGER AS $$
DECLARE
  v_portfolio_id UUID;
  v_total_units  NUMERIC;
  v_avg_price    NUMERIC;
BEGIN
  -- Tentukan portfolio_id dari row yang berubah
  IF TG_OP = 'DELETE' THEN
    v_portfolio_id := OLD.portfolio_id;
  ELSE
    v_portfolio_id := NEW.portfolio_id;
  END IF;

  -- Kalkulasi total units aktif
  -- BELI + SALDO_AWAL menambah, JUAL mengurangi
  SELECT
    COALESCE(
      SUM(CASE
        WHEN transaction_type IN ('BELI', 'SALDO_AWAL') THEN units
        WHEN transaction_type = 'JUAL'                  THEN -units
        ELSE 0
      END), 0
    )
  INTO v_total_units
  FROM asset_transactions
  WHERE portfolio_id = v_portfolio_id;

  -- Kalkulasi average buy price (DCA)
  -- Hanya dari BELI dan SALDO_AWAL — JUAL tidak masuk perhitungan DCA
  SELECT
    CASE
      WHEN SUM(CASE WHEN transaction_type IN ('BELI', 'SALDO_AWAL') THEN units ELSE 0 END) > 0
      THEN
        SUM(CASE WHEN transaction_type IN ('BELI', 'SALDO_AWAL') THEN total_amount ELSE 0 END) /
        SUM(CASE WHEN transaction_type IN ('BELI', 'SALDO_AWAL') THEN units ELSE 0 END)
      ELSE 0
    END
  INTO v_avg_price
  FROM asset_transactions
  WHERE portfolio_id = v_portfolio_id;

  -- Update portfolio stats
  UPDATE user_portfolios
  SET
    total_units       = v_total_units,
    average_buy_price = v_avg_price,
    -- Auto close portfolio jika total_units habis (semua dijual)
    closed_at = CASE
      WHEN v_total_units <= 0 THEN NOW()
      ELSE NULL
    END,
    updated_at = NOW()
  WHERE id = v_portfolio_id;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger untuk INSERT, UPDATE, DELETE di asset_transactions
CREATE TRIGGER trg_update_portfolio_stats
  AFTER INSERT OR UPDATE OR DELETE ON asset_transactions
  FOR EACH ROW EXECUTE FUNCTION update_portfolio_stats();

-- ==========================================
-- 3. VIEWS
-- Diletakkan di sini karena views membutuhkan tabel yang sudah ada
-- ==========================================

-- View: saldo wallet kalkulasi dari semua fiat_transactions
-- PEMASUKAN menambah, PENGELUARAN & TRANSFER_OUT mengurangi
CREATE OR REPLACE VIEW wallet_balances AS
SELECT
  w.id    AS wallet_id,
  w.name,
  w.user_id,
  w.currency,
  COALESCE(SUM(
    CASE
      WHEN ft.transaction_type = 'PEMASUKAN'  THEN ft.amount
      WHEN ft.transaction_type = 'PENGELUARAN' THEN -ft.amount
      -- TRANSFER: debit dari wallet asal, credit ke wallet tujuan
      WHEN ft.transaction_type = 'TRANSFER' AND ft.wallet_id = w.id     THEN -ft.amount
      WHEN ft.transaction_type = 'TRANSFER' AND ft.to_wallet_id = w.id  THEN  ft.amount
      ELSE 0
    END
  ), 0) AS balance
FROM wallets w
LEFT JOIN fiat_transactions ft ON (ft.wallet_id = w.id OR ft.to_wallet_id = w.id)
WHERE w.deleted_at IS NULL
GROUP BY w.id, w.name, w.user_id, w.currency;

-- View: harga terakhir tiap aset (dipakai getPortfolioWithDetails & getPortfolioSummary)
CREATE OR REPLACE VIEW latest_asset_prices AS
SELECT DISTINCT ON (asset_id)
  asset_id,
  price_per_unit AS latest_price,
  source,
  recorded_at
FROM asset_valuations
ORDER BY asset_id, recorded_at DESC;

-- View: total nilai aset user dikelompokkan per currency
CREATE OR REPLACE VIEW user_asset_value_by_currency AS
SELECT
  up.user_id,
  a.currency,
  SUM(lap.latest_price * up.total_units) AS total_value
FROM user_portfolios up
JOIN assets a ON a.id = up.asset_id
LEFT JOIN latest_asset_prices lap ON lap.asset_id = up.asset_id
WHERE up.closed_at IS NULL
  AND up.total_units > 0
GROUP BY up.user_id, a.currency;
